function animateNumber(element, start, end, duration) {
    const hasPlus = element.textContent.includes('+');
    const range = end - start;
    const increment = range / (duration / 16);
    let current = start;

    const timer = setInterval(() => {
        current += increment;
        if (current >= end) {
            current = end;
            clearInterval(timer);
        }
        element.textContent = Math.floor(current) + (hasPlus ? '+' : '');
    }, 16);
}
```

5 KALKULATORÓW z zaawansowanymi obliczeniami:


*** KALKULATOR 1: RAM ***

Wzór matematyczny:
baseRAM = 2 + (players/10 * 0.5) + (plugins/5 * 0.3)
baseRAM *= worldSize * serverType

Gdzie:
- players: liczba graczy
- plugins: liczba pluginów
- worldSize: mnożnik 1.0-2.0 zależnie od rozmiaru
- serverType: mnożnik 1.0-1.5 dla różnych typów serwera

Funkcje specjalne:
- Automatyczne generowanie flag JVM Aikar w zależności od RAM
- Trzy zestawy flag (dla 2-4GB, 6-10GB, 12+GB)
- Wizualizacja progress bar
- Opisowe rekomendacje


*** KALKULATOR 2: SLOTS (Max Graczy) ***

Wzór matematyczny:
cpuScore = cpuGhz * 10
ramScore = ram * 5
diskScore = diskType * 20
pluginPenalty = plugins * 0.5
totalScore = cpuScore + ramScore + diskScore - pluginPenalty
maxPlayers = floor(totalScore / 2)

Funkcje specjalne:
- Analiza wąskich gardeł (bottleneck analysis)
- System oceny dla każdego komponentu
- Rekomendacje upgrade'u
- Dynamiczne ostrzeżenia


*** KALKULATOR 3: CHUNKS ***

Wzór matematyczny:
viewChunksPerPlayer = (2*viewDistance + 1)²
totalViewChunks = viewChunksPerPlayer * onlinePlayers * worlds
simChunksPerPlayer = (2*simDistance + 1)²
totalSimChunks = simChunksPerPlayer * onlinePlayers * worlds

Funkcje specjalne:
- Obliczanie obciążenia w czasie rzeczywistym
- Progowanie (niskie/średnie/wysokie obciążenie)
- Automatyczne rekomendacje optymalizacji
- Kalkulacja procentowej redukcji obciążenia
- Kolorowy progress bar (zielony/pomarańczowy/czerwony)


*** KALKULATOR 4: KOSZTY ***

Wzór matematyczny:
baseCost = (ram * hostingType) + ddosProtection + backupCost

Gdzie:
- hostingType: 0 (domowy), 3 (shared), 6 (VPS), 12 (dedykowany)
- Dodatkowe usługi dodawane liniowo

Funkcje specjalne:
- Szczegółowy breakdown kosztów
- Obliczanie kosztów rocznych
- Personalizowane porady (tips)
- Obsługa hostingu domowego (0 zł)


*** KALKULATOR 5: TPS i Tick Speed ***

Wzory matematyczne:
tpsPercent = (currentTPS / 20) * 100
actualTicks = (normalTicks / 20) * currentTPS
realTime = normalTicks / currentTPS
delay = realTime - actualTime

clockTicksActual = currentTPS / redstoneClock
actualSpawnRate = (mobSpawns / 20) * currentTPS
growthPenalty = ((20 - currentTPS) / 20) * 100

Funkcje specjalne:
- Analiza wpływu TPS na wszystkie mechanizmy gry
- Obliczanie opóźnień dla konkretnych wartości ticków
- Wpływ na redstone, spawny, wzrost roślin, hoppery

function openLightbox(index) {
    currentImageIndex = index;
    const lightbox = document.getElementById('lightbox');
    const img = document.getElementById('lightbox-img');

    img.src = galleryImages[index].src;
    lightbox.classList.add('active');
    document.body.style.overflow = 'hidden';
}

function changeLightbox(direction, event) {
    event.stopPropagation();
    currentImageIndex += direction;

    // Wrap around
    if (currentImageIndex < 0) {
        currentImageIndex = galleryImages.length - 1;
    } else if (currentImageIndex >= galleryImages.length) {
        currentImageIndex = 0;
    }

    // Fade effect
    img.style.opacity = '0';
    setTimeout(() => {
        img.src = galleryImages[currentImageIndex].src;
        img.style.opacity = '1';
    }, 150);
}

// Keyboard navigation
document.addEventListener('keydown', function(event) {
    if (lightbox.classList.contains('active')) {
        if (event.key === 'Escape') {
            closeLightbox();
        } else if (event.key === 'ArrowLeft') {
            changeLightbox(-1, event);
        } else if (event.key === 'ArrowRight') {
            changeLightbox(1, event);
        }
    }
});
```
